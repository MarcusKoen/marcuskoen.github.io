# 1.0 Static XOR
```
#include <iostream>
#include <cstring>

using namespace std;

// XOR key
const char KEY = 0x55;

// Encrypted version of "secret123"
unsigned char encrypted[] = {
    's' ^ KEY,
    'e' ^ KEY,
    'c' ^ KEY,
    'r' ^ KEY,
    'e' ^ KEY,
    't' ^ KEY,
    '1' ^ KEY,
    '2' ^ KEY,
    '3' ^ KEY,
    0x00
};

void decrypt(char* out) {
    for (int i = 0; encrypted[i] != 0x00; i++) {
        out[i] = encrypted[i] ^ KEY;
    }
}

int main() {
    char input[50];
    char decrypted[50] = {0};

    cout << "Enter password: ";
    cin >> input;

    decrypt(decrypted);

    if (strcmp(input, decrypted) == 0) {
        cout << "Correct!" << endl;
    } else {
        cout << "Wrong!" << endl;
    }

    return 0;
}
```
